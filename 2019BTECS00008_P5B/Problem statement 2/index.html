Program To Find The Sum Of The Series 1 + 11 + 111 + 1111 + ... n terms.

Code

            #include 
            using namespace std;
             
            // Function for finding summation
            int summation(int n)
            {
                int sum = 0, j = 1;
                for (int i = 1; i <= n; i++) {
                    sum = sum + j;
             
                    // Appending a 1 at the end
                    j = (j * 10) + 1;
                }
             
                return sum;
            }
             
            // Driver Code
            int main()
            {
                int n = 4;
                cout << " " <<  summation(n);
                return 0;
            }

Sample input:

4

Sample output

1234

Variable used in the program

n sum i j
